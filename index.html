Deploy TWO Google Apps Scripts (one per sheet) and paste their URLs below:
1. Open Google Spreadsheet → Extensions → Apps Script
2. Replace all code with the script below → Save → Deploy → New Deployment → Web App → Execute as: Me → Who has access: Anyone

Apps Script Code — Select All & Copy:
function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow(['#','Date','Time','Symbol','Exchange','Direction', 'Entry ₹','SL ₹','Target ₹','Exit ₹','PnL ₹','Exit Type','Qty','Lots','Risk ₹','Capital ₹','R:R','Order Type','Status','Notes']); } sheet.appendRow([ data.id, data.date, data.time, data.symbol, data.exchange, data.direction, data.entry, data.sl, data.target, data.exitPrice||'', data.pnl||'', data.exitType||'', data.qty, data.lots, data.risk, data.capital, data.rr, data.orderType||'MARKET', data.status, data.notes ]); return ContentService.createTextOutput(JSON.stringify({status:'ok',row:sheet.getLastRow()})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
ℹ Open Trades URL receives new trades when placed. Closed Trades URL receives trades when they exit.
⚠ If CORS blocks: prefix URL with http://localhost:8010/proxy/
